-- Form Engine v2 proposal (separate from sql.txt)
-- This file is intentionally separated to avoid breaking existing DB schema.

CREATE TABLE form (
    form_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title VARCHAR(255) NOT NULL,
    description TEXT,
    is_template BOOLEAN NOT NULL DEFAULT FALSE,
    is_public BOOLEAN NOT NULL DEFAULT FALSE,
    is_paid BOOLEAN NOT NULL DEFAULT FALSE,
    price NUMERIC(12, 2) NOT NULL DEFAULT 0,
    status VARCHAR(20) NOT NULL DEFAULT 'DRAFT',
    version INT NOT NULL DEFAULT 1,
    published_at TIMESTAMP,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_by UUID NOT NULL REFERENCES user_account(user_id)
);

CREATE TABLE form_section (
    section_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    form_id UUID NOT NULL REFERENCES form(form_id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    order_index INT NOT NULL,
    allow_repeat BOOLEAN NOT NULL DEFAULT FALSE,
    repeat_label VARCHAR(120),
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (form_id, order_index)
);

CREATE TABLE form_question (
    question_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    section_id UUID NOT NULL REFERENCES form_section(section_id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    question_type VARCHAR(30) NOT NULL,
    is_required BOOLEAN NOT NULL DEFAULT FALSE,
    allow_repeat BOOLEAN NOT NULL DEFAULT FALSE,
    order_index INT NOT NULL,
    min_value NUMERIC(12, 2),
    max_value NUMERIC(12, 2),
    min_length INT,
    max_length INT,
    validation_pattern VARCHAR(255),
    validation_message VARCHAR(255),
    placeholder TEXT,
    helper_text TEXT,
    scale_min INT,
    scale_max INT,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (section_id, order_index)
);

CREATE TABLE form_question_option (
    option_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    question_id UUID NOT NULL REFERENCES form_question(question_id) ON DELETE CASCADE,
    content TEXT NOT NULL,
    score NUMERIC(12, 2) NOT NULL DEFAULT 0,
    order_index INT NOT NULL,
    trigger_logic JSONB,
    UNIQUE (question_id, order_index)
);

CREATE TABLE answer_session (
    session_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    form_id UUID NOT NULL REFERENCES form(form_id) ON DELETE CASCADE,
    patient_id UUID NOT NULL REFERENCES patient(patient_id),
    visit_id UUID REFERENCES visit(visit_id),
    status VARCHAR(20) NOT NULL DEFAULT 'DRAFT',
    source VARCHAR(20) NOT NULL DEFAULT 'PATIENT',
    started_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    submitted_at TIMESTAMP,
    last_saved_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    submitted_by UUID REFERENCES user_account(user_id),
    total_score NUMERIC(12, 2) NOT NULL DEFAULT 0
);

CREATE TABLE form_answer (
    answer_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID NOT NULL REFERENCES answer_session(session_id) ON DELETE CASCADE,
    question_id UUID NOT NULL REFERENCES form_question(question_id),
    option_id UUID REFERENCES form_question_option(option_id),
    repeat_index INT NOT NULL DEFAULT 0,
    value_text TEXT,
    value_number NUMERIC(12, 2),
    value_date DATE,
    value_boolean BOOLEAN,
    value_json JSONB,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_form_public_status ON form (is_public, status);
CREATE INDEX idx_form_section_form_order ON form_section (form_id, order_index);
CREATE INDEX idx_form_question_section_order ON form_question (section_id, order_index);
CREATE INDEX idx_answer_session_patient_form ON answer_session (patient_id, form_id);
CREATE INDEX idx_form_answer_session_question ON form_answer (session_id, question_id);
